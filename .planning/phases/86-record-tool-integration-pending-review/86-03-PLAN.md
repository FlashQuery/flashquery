```typescript
function formatReconciliationSummary(summary: ReconciliationActionSummary): string {
  const parts: string[] = [];
  if (summary.autoTracked > 0) parts.push(`Auto-tracked ${summary.autoTracked} new document(s)`);
  if (summary.archived > 0) parts.push(`Archived ${summary.archived} record(s) (documents missing or disassociated)`);
  if (summary.resurrected > 0) parts.push(`Resurrected ${summary.resurrected} record(s)`);
  if (summary.pathsUpdated > 0) parts.push(`Updated paths for ${summary.pathsUpdated} moved document(s)`);
  if (summary.fieldsSynced > 0) parts.push(`Synced fields on ${summary.fieldsSynced} modified document(s)`);
  return parts.length > 0 ? `\nReconciliation: ${parts.join('. ')}.` : '';
}
```

```typescript
async function queryPendingReview(
  pluginId: string,
  instanceName: string
): Promise<Array<{ fqc_id: string; table_name: string; review_type: string; context: unknown }>> {
  const supabase = supabaseManager.getClient();
  const { data } = await supabase
    .from('fqc_pending_plugin_review')
    .select('fqc_id, table_name, review_type, context')
    .eq('plugin_id', pluginId)
    .eq('instance_id', instanceName);
  return data ?? [];
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP caller → record tools | AI callers provide `plugin_id`, `plugin_instance`, `table_name` — treated as untrusted |
| Record tool → reconcilePluginDocuments | Plugin ID comes from the MCP call, passed to reconciliation engine |
| documents.ts → plugin registry | Path from MCP caller compared against plugin-declared folder claims |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-86-07 | Denial of Service | Reconciliation preamble adds DB roundtrip to every record tool call | accept | Staleness cache (30s) limits DB calls to one per 30s per plugin/instance. Reconciliation is the intended design per v2.8 goal. |
| T-86-08 | Information Disclosure | Reconciliation warning text in response may expose internal error details | accept | Warning text comes from caught exceptions in controlled code paths; no user secrets in reconciliation stack |
| T-86-09 | Elevation of Privilege | `access: read-only` guardrail is warning-only; AI caller could intentionally write to read-only folder | accept | Hard enforcement explicitly deferred to post-v2.8 per REQUIREMENTS.md Out of Scope section. Warning is sufficient for v2.8 operational transparency. |
| T-86-10 | Tampering | `unregister_plugin` deletes all pending reviews — could be called maliciously to clear another plugin's data | accept | Plugin scoping via `.eq('plugin_id', plugin_id)` limits deletion to the specified plugin. MCP auth layer is the outer defense for authorization. |
</threat_model>
